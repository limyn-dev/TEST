[string]   $P = $script:STP.PROPS["PACKAGE_ID"]                                     #
[string[]] $A = $P.Replace(" - ", ";").Split(";")                                   #
[string]   $K = "$($script:TGR)\$($A[0]) - $($A[1])"                                # Registry key
#                                                                                   -----------------------------------
Lv $script:STP.TEMPLATE "Clean package TAG [$P]"                                    # Clean TAG
RP $K $P -Force                                                                     #
#                                                                                   -----------------------------------
